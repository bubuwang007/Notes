## 定义符号及函数

~~~
syms a b c abc x_1; %定义多个符号，用空格隔开
aa = [a,b]; %符号可以创建矩阵

f = a^2+b; %定义符号函数

syms f1(x,y) %定义符号函数
f1 = a^2+b; %定义表达式

eq1 = f == f1; %定义符号等式

f2 = x^2;
ezplot(f2); %当表达式只有一个变量，使用该函数绘图
~~~



## 符号运算

~~~
syms x

f = 2*x^2 +x^2;
collect(f); %合并表达式中同幂次项 3*x^2

f1 = x*(1+x);
expand(f1); %展开表达式

f2 = x^3-1;
factor(f2); %因式分解,返回符号矩阵

f3 = 1/(x^4-1) + 1/(x^3-1);
[m,n] = numden(f3); %通分,需要一个1*2矩阵接受结果，m为分子，n为分母

f4 = sin(x)^2 + cos(x)^2
simplify(f4); %化简，f4 = 1

f5 = 2*x^2 +1;
pretty(f5); %按数学方式书写

f6 = sin(x);
finverse(f6); %求反函数

~~~
